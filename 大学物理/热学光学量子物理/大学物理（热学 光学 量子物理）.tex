\section{热学}
\subsection{理想气体基本公式}
\begin{itemize}
    \item 理想气体物态方程：\keyformula{pV=\frac{m}{M}RT}
    \item 压强微观表达式：$p = \frac{2}{3} n \overline{\varepsilon}_{\text{kt}} = \frac{2}{3} n \left( \frac{1}{2} m_0 \overline{v^2} \right)$
    \item 物态方程另一形式：$p = nkT$
    \item 玻尔兹曼常量与气体常量关系：$k=\frac{R}{N_{\text{A}}}$
\end{itemize}

\textbf{符号说明：}\\
\symb{$p$}{压强（Pa）}
\symb{$V$}{体积（m³）}
\symb{$T$}{热力学温度（K）}
\symb{$m$}{气体质量（kg）}
\symb{$M$}{摩尔质量（kg/mol）}
\symb{$n$}{分子数密度（m⁻³）}
\symb{$R$}{气体常量（8.31 J/(mol·K)）}
\symb{$k$}{玻尔兹曼常量（$1.38\times10^{-23}$ J/K）}
\symb{$m_0$}{分子质量（kg）}
\symb{$N_{\text{A}}$}{阿伏伽德罗常量（$6.02\times10^{23}$ mol⁻¹）}

\subsection{速率分布函数}
\begin{itemize}
    \item 定义：$\frac{\mathrm{d}N}{N_0} = f(v)\mathrm{d}v$ 或 $f(v) = \frac{\mathrm{d}N}{N_0 \mathrm{d}v}$
    \item 麦克斯韦速率分布：
    \keyformula{f(v) = 4\pi \left( \frac{m_0}{2\pi kT} \right)^{\frac{3}{2}} e^{-\frac{m_0 v^2}{2kT}} v^2}
    \item 归一化条件：$\int_{0}^{\infty} f(v) \mathrm{d}v = 1$
\end{itemize}

\textbf{符号说明：}\\
\symb{$f(v)$}{速率分布函数（s/m）}
\symb{$m_0$}{分子质量（kg）}
\symb{$N_0$}{总分子数（个）}

\subsection{分子动能与自由度}
\begin{itemize}
    \item 平均平动动能：\keyformula{\overline{\varepsilon}_{\text{kt}} = \frac{1}{2} m_0 \overline{v^2}=\frac{3}{2}kT}
    \item 自由度表格：

    \item 分子的平均动能：$\overline{\varepsilon}_{\text{k}} = \frac{i}{2} kT$（$i$为分子的自由度数）
    \item 理想气体内能：\keyformula{E=\frac{m}{M}\frac{i}{2}RT}

\subsection{平均自由程与碰撞频率}
\begin{itemize}
    \item 分子的碰撞截面：$\sigma = \pi d^2$
    \item 平均碰撞频率：$\bar{z} = \sqrt{2} \sigma \bar{v} n = \sqrt{2} \pi d^2 \bar{v} n$
    \item 平均自由程：
    \keyformula{\overline{\lambda} = \frac{\bar{v}}{\bar{z}} = \frac{1}{\sqrt{2} \sigma n} = \frac{kT}{\sqrt{2}\pi d^2p}}
\end{itemize}

\symb{$n$}{分子数密度（m⁻³）}

\subsection{热力学过程}
\subsubsection{热容相关}
\begin{itemize}
    \item 摩尔定容热容：$C_{V,\text{m}} = \left( \frac{\mathrm{d}Q}{\mathrm{d}T} \right)_V = \frac{\mathrm{d}E}{\mathrm{d}T} = \frac{i}{2} R$
    \item 摩尔定压热容：$C_{p,\text{m}} = \left( \frac{\mathrm{d}Q}{\mathrm{d}T} \right)_p = \frac{\mathrm{d}E}{\mathrm{d}T} + R$
    \item 迈耶公式：\keyformula{C_{p,\text{m}} = C_{V,\text{m}} + R}
    \item 比热容比：$\gamma = \frac{C_{p,\text{m}}}{C_{V,\text{m}}}$
\end{itemize}

\symb{$C_{V,\text{m}}$}{摩尔定容热容（J/(mol·K)）}

\subsection{热机与熵}
\begin{itemize}
    \item 热机效率：\keyformula{\eta = \frac{W}{Q_1} = 1 - \frac{Q_2}{Q_1}}
    \item 制冷系数：$\varepsilon = \frac{Q_2}{W} = \frac{Q_2}{Q_1 - Q_2}$
    \item 熵：$S = k\ln \Omega$
    \item 熵变：$\Delta S = k\ln \frac{\Omega_2}{\Omega_1} > 0$，$\mathrm{d}S = \frac{\mathrm{d}Q}{T}$

\end{itemize}

\section{光学}
\subsection{半波损失}
\corepoint{半波损失}
\begin{itemize}
    \item 定义：光从光疏→光密介质反射时，相位突变 $\pi$（光程差加 $\lambda/2$），折射光无半波损失；
    \item 条件：仅反射光、光疏→光密入射（$n_1 < n_2$）；
    \item 影响：干涉加强/减弱条件反转（薄膜/劈尖/牛顿环分析关键）。
\end{itemize}

\textbf{符号说明：}\\
\symb{$\lambda$}{入射光波长（m）}
\symb{$n$}{介质折射率（无量纲）}

\subsection{光的干涉}
\subsubsection{杨氏双缝}
\begin{itemize}
    \item 光程差：$\delta = r_2 - r_1 \approx \frac{xd}{D}$
    \item 明纹：$\delta = \pm k\lambda \Rightarrow x = \pm k \frac{D\lambda}{d}\quad (k=0,1,2,\dots)$
    \item 暗纹：$\delta = \pm (2k-1)\frac{\lambda}{2} \Rightarrow x = \pm (2k-1)\frac{D\lambda}{2d}$
    \item 条纹间距：\keyformula{\Delta x=\frac{D\lambda}{d}}
\end{itemize}

\textbf{符号说明：}\\
\symb{$d$}{缝间距（m）}
\symb{$D$}{屏距（m）}
\symb{$x$}{条纹位置（m）}
\symb{$k$}{条纹级次（无量纲）}

\subsubsection{薄膜等倾干涉}
\begin{itemize}
    \item 光程差：
    \keyformula{\delta = 2e \sqrt{n_2^2 - n_1^2 \sin^2 i} + \frac{\lambda}{2}}
    \item 明纹：$2e \sqrt{n_2^2 - n_1^2 \sin^2 i} = \left( k - \frac{1}{2} \right)\lambda$
    \item 暗纹：$2e \sqrt{n_2^2 - n_1^2 \sin^2 i} = k\lambda$
\end{itemize}

\textbf{符号说明：}\\
\symb{$e$}{薄膜厚度（m）}
\symb{$n_1/n_2$}{介质折射率（无量纲）}
\symb{$i$}{入射角（rad）}

\subsubsection{劈尖干涉}

\subsubsection{牛顿环}
\begin{itemize}
    \item 光程差：$\delta = 2e + \frac{\lambda}{2}$（$e \approx \frac{r^2}{2R}$）
    \item 明环：$r_k = \sqrt{(k-\frac{1}{2})\lambda R}$
    \item 暗环：$r_k = \sqrt{k\lambda R}$
    \item 曲率半径测量：
    \keyformula{R=\frac{d_{k+m}^{2}-d_{k}^{2}}{4 m \lambda}}
\end{itemize}

\textbf{符号说明：}\\
\symb{$r_k$}{环半径（m）}
\symb{$R$}{透镜曲率半径（m）}
\symb{$d_k$}{第$k$圈暗环直径（m）}

\subsection{光的衍射}
\subsubsection{单缝衍射（夫琅禾费）}
\corepoint{单缝衍射}
\begin{itemize}
    \item 半波带法：最大光程差 $\Delta = a\sin\theta$，半波带数 $N = \frac{2a\sin\theta}{\lambda}$
    \item 暗纹：$a\sin\theta = \pm k\lambda\quad (k=1,2,3,\dots)$
    \item 中央明纹：半角宽度 $\theta_0 \approx \frac{\lambda}{a}$，角宽度 $\Delta\theta_0 \approx \frac{2\lambda}{a}$
    \item 相对光强：$\frac{I}{I_0} = \left( \frac{\sin\alpha}{\alpha} \right)^2$（$\alpha = \frac{\pi a\sin\theta}{\lambda}$）
\end{itemize}

\textbf{符号说明：}\\
\symb{$a$}{缝宽（m）}
\symb{$\theta$}{衍射角（rad）}
\symb{$\Delta\theta_0$}{中央明纹角宽度（rad）}

\subsubsection{光栅衍射}
\begin{itemize}
    \item 光栅常量：$d=a+b$（$a$ 缝宽，$b$ 刻痕宽）
    \item 光栅方程（主极大）：\keyformula{d\sin\varphi = \pm k\lambda\quad (k=0,1,2,\dots)}
    \item 缺级条件：$k = k'\frac{d}{a}\quad (k'=\pm1,\pm2,\dots)$
\end{itemize}

\textbf{符号说明：}\\
\symb{$d$}{光栅常量（m）}
\symb{$\varphi$}{衍射角（rad）}
\symb{$k$}{主极大级次（无量纲）}

\subsubsection{光学仪器的分辨本领}
\begin{itemize}
    \item 瑞利判据：\keyformula{\delta\theta = 1.22 \frac{\lambda}{d}}
    \item 分辨本领：$R = \frac{1}{\delta\theta} = \frac{d}{1.22\lambda}$
\end{itemize}

\textbf{符号说明：}\\
\symb{$\lambda$}{入射光波长（m）}
\symb{$d$}{透镜/圆孔直径（m）}

\subsection{光的偏振}
\subsubsection{马吕斯定律}
\keyformula{I_2 = I_1 \cos^2\alpha}（自然光入射：$I_1 = \frac{I_0}{2}$）

\textbf{符号说明：}\\
\symb{$I_0$}{入射光强（W/m²）}
\symb{$\alpha$}{偏振方向夹角（rad）}

\subsubsection{布儒斯特定律}
\keyformula{\tan i_0 = \frac{n_2}{n_1}}

\subsubsection{康普顿效应}

    \item 康普顿波长：$\lambda_{\text{C}}=\frac{h}{m_0c}=2.43\times10^{-12}$ m

\subsubsection{德布罗意波}

    \item 物质波波长（相对论）：\\
      $\lambda = \frac{hc}{\sqrt{E_{\text{k}}^2 + 2m_0c^2E_{\text{k}}}}$
    \item 玻尔轨道量子化：$2\pi r = n\lambda \Rightarrow mvr = n\hbar$

\subsection{量子力学基础}
\subsubsection{波函数}
\corepoint{波函数核心性质}
\begin{itemize}
    \item 概率密度：$w = |\Psi|^2 = \Psi^* \cdot \Psi$
    \item 归一化条件：$\iiint_{-\infty}^{+\infty} |\Psi|^2 \mathrm{d}x\mathrm{d}y\mathrm{d}z = 1$
    \item 标准条件：单值、有限、连续
\end{itemize}

\subsubsection{不确定关系}
\corepoint{不确定性原理核心应用}
\begin{itemize}
    \item 位置-动量不确定关系：$\Delta x \Delta p_x \ge \frac{\hbar}{2}$
    \item 时间-能量不确定关系：$\Delta E \cdot \Delta t \ge \frac{\hbar}{2}$
    \item 计算技巧：结合德布罗意关系 $p=\frac{h}{\lambda}$，由动量相对不确定度求位置不确定度
\end{itemize}
\keyformula{\Delta x \Delta p_x \ge \frac{\hbar}{2}}

\subsubsection{薛定谔方程与一维势阱}
\corepoint{一维无限深势阱核心考点（含宽度2a特殊情形）}
\begin{itemize}
    \item 一维定态薛定谔方程：\\
      $-\frac{\hbar^2}{2m}\frac{\mathrm{d}^2\psi}{\mathrm{d}x^2} + V(x)\psi = E\psi$
    
    % 基础情形：宽度为a的势阱（0 < x < a）
    \item 宽度为$a$的一维无限深势阱（$0 < x < a$）：
      - 归一化波函数：$\psi_n(x) = \sqrt{\frac{2}{a}} \sin\left(\frac{n\pi x}{a}\right)$
      - 能量本征值：$E_n=\frac{n^2h^2}{8ma^2}\quad (n=1,2,3,\dots)$
    
    % 补充：宽度为2a的势阱（对应题目考点）
    \item 宽度为$2a$的一维无限深势阱（$0 < x < 2a$）：
      - 归一化波函数：$\psi_n(x) = \begin{cases}
        \sqrt{\dfrac{1}{a}} \sin\left(\dfrac{n\pi x}{2a}\right), & 0 < x < 2a \\
        0, & x \le 0 \text{ 或 } x \ge 2a
        \end{cases}$
      - 能量本征值：
      \keyformula{E_n = \frac{n^2 \pi^2 \hbar^2}{8 m a^2} = \frac{n^2 h^2}{32 m a^2}\quad (n=1,2,3,\dots)}
    
    % 能级跃迁（对应题目第二问）
    \item 能级跃迁与光子频率：
      - 第二激发态（$n=3$）→ 基态（$n=1$）能量差：\\
        $\Delta E = E_3 - E_1 = \frac{\pi^2 \hbar^2}{m a^2}$
      - 辐射光子频率：
      \keyformula{\nu = \frac{\Delta E}{h} = \frac{\pi \hbar}{2 m a^2}}
    
    % 经典极限（对应题目第三问）
    \item 相邻能级差与经典极限：
      - 相邻能级差：$\Delta E_n = E_{n+1} - E_n = \frac{(2n+1)\pi^2 \hbar^2}{8 m a^2}$
      - 经典极限：$n \to \infty$时，$\frac{\Delta E_n}{E_n} = \frac{2n+1}{n^2} \to 0$，\\
        能级连续，量子结果趋近经典理论
\end{itemize}

\symb{$\hbar$}{约化普朗克常量（$\hbar = h/(2\pi)$）}
